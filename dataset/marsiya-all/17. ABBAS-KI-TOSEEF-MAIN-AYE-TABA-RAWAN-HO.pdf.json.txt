بسم الله الرحمن الرحیم

میرانیس اور ان کے اخلاف

مرثیہ میر انیسؔ

1
عباس کی توصیف میں اس طبع رواں ہو
اے ناطقہ سحر بیاں صرف بیاں ہو
اے لوح خدا بخش شبستان جہاں ہو
اے خامہ اعجاز رقم نور فشاں ہو
لکھ مدح و ثنا بازوئے شاہِ شہدا کی
تصویر دکھا دے پر شیر خدا کی

2
اس شکل سے تحریر ہو اوصاف سراپا
صورت گر اعجازِ رقم جن پہ ہوں شیدا
آنکھوں کو نظر آئے بعینہ وہی نقشا
آئینہ تصویر ہو ہر مصرعۂِ زیبا
غل بزم میں ہوشہ کے علمدار کو دیکھا
دیکھا جو اسے حیدر کرار کو دیکھا

3
ہر لفظ ہو آئنہ چین کے لئے اوصاف
بیت ابروں کی مطلع خورشید سے ہو صاف
توصیف مژہ سن کے کہیں صاحبِ انصاف
باریکی مضمون کا ہر اک حرف ہے کشّاف
یہ عین فصاحت ہے کہ آنکھوں کا بیاں ہو
خون اشک کی جا دیدۂ مردم سے رواں ہو

4
خورشید سے تھے جلوہ نما عارضِ انور
خط گرد تھا یوں ماہ ہو جوں ہالیکے اندر
لب برگ گلِ خلد دہن چشمہؑ کوثر
دنداں وہ جنہیں پہنچے نہ الماس نہ گوہر
خوروں نے بھی اس حسن کی گردن نہیں دیکھی
یوں شمع کسی بزم میں روشن نہیں دیکھی

5
حمیزہ کی طرح زید میں یکتا تھا وہ خوشخو
تیار تھے جعفر کی طرح ساعد و باز
ہاشم کی طرح ذی حشم و خوش قد و خوشرو
ساری اسد اللہ کی شان اور وہی خوبو
مانند حسن محسن عالم وہ سخی تھا
اور طبع میں احسان حسین ابن علی تھا

6
خوبی میں ہر اک عضو کا تھا طور نرالا
تھا ہا شمیوں کے وہ محلّے کا اُجالا
آفت میں تھا ہردم سپر سید والا
پالا تھا جوانان جہاں سے قد بالا
آئینہ تھا اس صاحب شمشیر کا سینا
دل باپ کا حمزہ کا جگر شیر کا سینا

7
حیدر کی طرح جسم میں تھا زورِ خدا داد
لاکھوں سے لڑا زمیں ہے چاہی نہیں امداد
تھا حُسن میں بھی منتخبِ عالم ایجاد
چهره گل شاداب تو قد غیرت شمشاد
ثابت قدمی ختم تھی صفدر کے قدم پر
پڑتا تھا قدم فاتح خیبر کے قدم پر

8
اس حسن پہ اس زور پہ تھا حلم بھی ایسا
منھ سے نہ کبھی حرف تکبّر کا نکالا
گر سامنے تعریف شجاعت کوئی کرتا
عباس جھکا لیتے تھے شرما کے سر اپنا
فرماتے تھے میں کیا ہوں یہ رتبہ میرا کب ہے
یہ سب شہ والا کی غلامی کا سبب ہے

9
تھا زہر میں وہ فخر جہاں ثانی حیدرؑ
پابند گئی شاہ تھی یا طاعت داور
تکبیر سے تہلیل سے فرصت تھی نہ دم بھر
گر حمد خدا نعت نبی گاہ زباں پر
آرائش دنیا کی نہ دولت کی طلب تھی
آٹھوں پہر اللہ سے رحمت کی طلب تھی

10
کرتے تھے علی جیسے پیمبرؑکی اطاعت
عباس نے کی ویسے برادر کی اطاعت
چھوٹوں کی طرح کرتے تھے سب گھرکی اطاعت
قاسم کی اطاعت علی اکبر کی اطاعت
فرماتے تھے یہ فاطمہ کے لخت جگر ہیں
آقا میرے یہ بھی ہیں کہ آقا کے پسر ہیں

11
تشویش میں گر فاطمہؑ کے لال کو پاتے
بے ساختہ اشک آنکھوں سے عباسؑ بہاتے
جب پیار سے سرور انھیں چھاتی سے لگاتے
آداب سے یہ قدموں پہ سر اپنا جھکاتے
رنج شہ دیں مرگ کا پیغام تھا ان کو
شپیرؑ کے آرام سے آرام تھا ان کو

12
تشریف کہیں گھر سے جو لیجاتے تھے سرور
ہمراہ یہ ہو لیتے تھے ہتھیار لگا کر
مسند پہ اگر بیٹھتا تھا وہ شہ صفدر
رومال ہلاتے تھے یہ جا کر عقبِ سر
شاہی سے غلامی فزوں جانتے تھے وہ
رہے شہ کونین کے بہیچانتے تھے وہ

13
تھا سبط نبیؐ کی بھی محبت کا یہ عالم
بھائی کو نہ آنکھوں سے جدا کرتے تھے اکرم
جب بیٹھے تو باہم جو گئے گھر سے تو باہم
گر خوش وہ تو یہ خوش انھیں غم تو انھیں غم
کیوں مرتبے اعلانہ ہوں عباس علی کے
عاشق بھی ہیں معشوق بھی ہیں سبطِ نبیؐ کے

14
غازی کی شجاعت کی ہر اقلیم میں تھی دھوم
تھرّاتا تھا دہشت سے عراق و عرب و روم
تھا رعب عدالت سے ستم خلق سے معدوم
ظالم کو یہ تھا خوف کہ بن جاتا تھا مظلوم
بے جان ہو وہ آزردہ کرے دل جو کسی کا
چل سکتا نہ تھا زور ضعیفوں پہ قوی کا

15
مرنے پہ بھی غازی کی شجاعت ہے نمودار
رہتا ہے سدا شہ کے محبّوں کا مددگار
ہوتے ہیں جو زائر کسی آفت میں گرفتار
تھراتا ہے اس وقت مزارِ شہِ ابرار
مانند علیؑ بہر مدد آتے ہیں عباسؑ
زائر کو مصیبت سے چھڑا جاتے ہیں عباسؑ

16
ایک چھوٹی سی بستی ہے مسیب لب دریا
واں سے کئی فرسخ ہے مزار شہ والا
مسلم کے تیموں کے بھی روضے ہیں اسی جا
اس بستی کا سردار خوارج میں کوئی تھا
ظالم کو زبس دشمنی آل عبا تھی
زوّاروں پہ بدعت تھی محبتوں پہ خفا تھی

17
منظور تھا جاوے نہ کوئی روضہ شہ پر
کہتا تھا ہر اک سے بخشونت وہ ستم گر
حج چھوڑ کے جاتے ہو سوئے روضہ سرور
زردے لو جو پہلے تو چڑھو کشتی کے اندر
بے اسکے قدم یاں سے بڑھانانہ ملے گا
درہم جو نہ دے گا اسے جانا نہ ملے گا

18
بدعت یہ کمر باندھے تھا وہ ظالم غدار
کیا بولتے معلوم تھے مظلوم کے زوار
زر دیتے تھے دریا سے اتر جاتے تھے زردار
جو ہوتے تھے محتاج وہ رہ جاتے تھے ناچار
غربت پسنہ کچھ ان کی ترس کھاتا تھا ظالم
خدام سے کوڑے انھیں لگواتا تھا ظالم

19
تھا ہند سے اک قافلہ زوّار کا جاتا
ان سب میں تھی اک زائرہ سید والا
محتاج تھی ایسی کہ نہ کچھ پاس تھا اصلا
طے منزلیں کرتی تھی ضعیفی میں پیادہ
نعلین جو پاؤں میں نہ تھے کانٹے گڑے تھے
چھالے کہیں تلوؤں میں کہیں زخم پڑے تھے

20
جز رعشہ کوئی ہاتھ پکڑتا تھا نہ آکر
جس وقت اٹھاتی تھی قدم کانپتا تھا سر
تھا طے منازل سے ورم قدموں کے اوپر
پر شوق زیارت میں چلی جاتی تھی دن بھر
دل جانب سپّر تھا خالق به نظر تھی
نہ پاؤں کا تھا ہوش نہ کچھ سر کی خبر تھی

21
فرسودہ بدن عالم پیری سے کمر خم
دن بھر چلی جاتی تھی عصا ٹیکتی باہم
جا بیٹھی کسی نخل تلے چڑھ گیا گردم
پھر اٹھ چلی ساتھ بڑھا قافلہ جس دم
تھکتی تھی نہ وہ لیکے مدد شاہ نجف سے
شوق اس کی طرف سے تھا کشش شہ کی طرف سے

22
کہتا تھا ترس کھا کے کوئی شہ کا جو زوار
ہم چلتے ہیں پیدل تو کجا دے میں ہوا اسوار
طے اب نہیں ہو سکنے کا یہ وادی پر خار
کہتی تھی میرے واسطے راحت ہو یہ آزار
پیدل ہی چلوں گی حسنات اس میں بڑے ہیں
کانٹے میری شہزادی کے تلوے میں گڑے ہیں

23
کہتا تھا تمازت میں کوئی آنکے گر پاس
مر جائیگی اس دھوپ میں تو ہے ہمیں وسواس
پیاسی ہے مشکیزہ ہے موجود بجھا پیاس
رو کر یہ بیاں کرتی تھی وہ بے کس و بے آس
کیا خاک میں پانی پیوں رونیکی یہ جا ہے
دو دن یہی پانی نہیں آقا کو ملا ہے

24
منزل پہ مسیّب کی جو وہ قافلہ پہونچا
جا اترے تھکے ماندے مسافر لب دریا
بستر کیا ریتی پہ ضعیفہ نے بھی اک جا
سب رات کو سوئے اُسے نیند آئی نہ اصلا
تھا شوق میں آرام کہاں دلکو ضزیں کے
تڑپا کی تصوّر میں مزار شہ دیں کے

25
ناگاہ عیاں ہونے لگے صبح کے آثار
اور دی یہ صدا قافلہ باشی نے کہ ہشیار
پڑھ پڑھ کے نماز سحری شاہ کے زوار
دریا سے اترنے کو چڑھے کشتی پہ یک بار
کس شوق سے ان کانپتے قدموں کو بڑھا کر
بیٹھی وہ ضعیفہ بھی اسی کشتی میں جا کر

26
واں حاکم اظلم کے ملازم بھی تھے موجود
نہ خوف خدا ان کو نہ اندیشہ معبود
زواروں سے جب کر چکے حاصل زر مقصود
اس بیوہ کے پاس آکے کہنے لگے مردود
کیا بیٹھی ہے تو زانو نپہ گردن کو جھکا کر
جانا ہے تو جلدی زر محصول ادا کر

27
یہ سنتے ہی وہ زائرہ سید ابرار
گھبرا کے لگی کہنے میں ہوں بے کس و ناچار
محتاج ہوں عاجز ہوں نہ درہم ہیں نہ دینار
دنیا میں نہیں مجھ سے زیادہ کوئی نادار
حاضر ہے یہ لے تو میرے پاس اور تو کیا ہے
واللہ یہی ایک عصا اور ردا ہے

28
آزار نہ دو رحم کرو حال پہ میرے
کیا دوں تمہیں خود کرتی ہوں میں فاقے پہ فاقے
اس راہ میں دینار میرے پاس جو ہوتے
جانے کی خوشی میں تمہیں دیدیتی میں پہلے
گو عالم غربت میں ہوں بدتر فقرا سے
پر دل ہے غنی الفت شاہ شہدا سے

29
بیدینوں کو جس دم یہ ضعیفہ نے سُنایا
نام شہ دیں سنتے ہی غصہ انھیں آیا
کوڑا بہ ستم پشت ضعیفہ پہ لگایا
اور کھینچ کے سب نے اُسے کشتی سے گرایا
یاں سب نے جو اس عاجزہ بے کس پہ جفا کی
تھرّا گئی واں قبر شہ کرب وبلا کی

30
کچھ رتبہ ضعیفہ کا نہ بیدینوں نے جانا
مجھے حسنہ زائرہ شہ کا رلانا
ثابت ہوا ان پہ جو ضعیفہ کا نہ جانا
ملاحوں نے کشتی کو کیا واں سے روانا
صدمہ یہ ہوا مومنہ پاک کے اوپر
اک آہ کی اور بیٹھ گئی خاک کے اوپر

31
مایوس ہوئی جانے سے جب زائرہ پاک
فریاد یوں کی طرح گریباں کو کیا چاک
منھ روضہ سرور کی طرف کر کے وہ غمناک
چلائی کی مولا میری محنت ہوئی سب خاک
کشتی سے اتارا ہے مجھے داد کو پہونچو
اے نوح غریباں مری فریاد کو پہونچو

32
اے بیوؤں کے وارث تری دیتی ہوں دہائی
یہ دکھ یہ ضعیفی یہ میری آبلہ پائی
کس رنج و مشقت سے میں تھی یہاں تلک آئی
سب خاک میں محنت میری اعدا نے ملائی
دریا سے اترنے میں نہیں پاتی ہوں آقا
محروم زیارت سے رہی جاتی ہوں آقا

33
گہ رو کے وہ چلاتی تھی اس طرح بصد یاس
زوّاروں کا بھائی کے بہت آپکو ہے پاس
آفت میں گرفتار ہوں میں بیکس و بے آس
آؤ میری امداد کو یا حضرت عباسؑ
ناچار ہوں کشتی تلک اب جا نہیں سکتی
دس کوس پہ روضہ ہے پہ میں آ نہیں سکتی

34
روئی جو وہ بیکس یہ بیاں کر کے بصد درد
ناگاہ اٹھا سامنے سے اک طبق گرد
دیکھا جو ضعیفہ نے ادھر بھر کے دم سرد
گھوڑے پہ نظر آیا اسے ایک جواں مرد
انداز شجاعت تو سراپا سے عیاں تھا
پر زیر نقاب اسکا رخ پاک نہاں تھا

35
بولا یہ ضعیفہ کے قریب آکے وہ اسوار
اے زائرہ کیا تجھ پہ مصبیت ہے کر اظہار
رو رو کے یہ کہنے لگی وہ بیکس و ناچار
دریا سے اتر نے نہیں دیتے ہیں جفا کار
بتیاب ہوں جوں ماہی بے آب زمیں پر
جانا نہیں ملتا ہے مزار شہ دیں پر

36
یہ رنج بڑھاپے میں کئے میں نے گوارا
جز پنجتن پاک نہ تھا کوئی سہارا
یہاں آئی تو جلا دوں نے کشتی سے اتارا
آقا کا لیا نام تو کوڑا مجھے مارا
روؤں گی میں جسدم یہ ستم یاد کروں گی
روضے پہ جو پونچوں گی تو فریاد کروں گی

37
تنہائی کی آفت میں ہوں میں ششدر و حیراں
دھڑکا ہے کہ پہونچائے گا اب کون مجھے واں
دریا یہ خطرناک یہ ریت اور یہ میدان
شب کو نہیں بچنے کی درندوں سے مری جاں
ہاتھ آئی زیارت نہ شہ جن و بشر کی
گھر سے بھی چھٹی اب نہ ادھر کی نہ ادھر کی

38
سنکر سخن زائرہ سرور زیجا
اسوار نے کھینچی دل غمدیدہ سے اک آہ
فرما یا ضعیفہ سے کہ آ چل میرے ہمراہ
اب یہ نہیں مقدور کہ روکیں تجھے بدخواہ
ہے امن میں طوفان سے محب ابن علی کا
جس کشتی میں تو ہے وہ سفینہ ہے بنی کا

39
اس مشردے کو سنتے ہی نہایت وہ ہوئی شاد

عباس ہی کیا مجھکو جو خالق نہ اٹھاتا
میں تجھ پہ فدا ہونے کو میدان میں جاتا
چھاتی پہ علی بدلے تری برچھیاں کھاتا
آگے ترے خنجر سے گلا اپنا کٹاتا
خوش ہوں کہ شریک شہدا ہوتا ہے عباس
بابا کے عوض تجھ پہ فدا ہوتا ہے عباس

77
سن سن کے یہ حیدر کے بیاں روتے تھے سرور
اور معرکہ آرا تھا علمدار دلاور
جب کر گئے دریا سے کنارا وہ ستم گر
رہوار کو تب ڈال دیا نہر کے اندر
پانی پہ توجہ بھی نہ کی بھر کریم نے
پر آب کیا مشک کو سقائے حرم نے

78
تھی در پہ سکینہ جو کھڑی آس لگائے
چلاتی تھی کیا ہے جو چچاپھر کے نہ آئے
مژدہ کوئی عباس کے آنے کا سنائے
نزدیک ہے اب جان میری پیاس سے جائے
بہلانے کو سب کہتے ہیں اب آئے گا پانی
مر جاؤں گی میں نہر سے تب آئے گا پانی

79
پاس آن کے ڈیوڑھی کے یہ کہنے لگے اکبر
لو بھینا پھرے نہر سے عباس دلاور
مشکیزہ بھی سیراب کیا فوج سے لڑکر
پیاسی تھیں بہت رحم خدا نے کیا تم پر
اب چاند سی صورت تمہیں کھلاتے ہیں عباس
لڑتے ہوئے اعدا سے چلے آتے ہیں عباس

80
خوش ہو کے سکینہ نے کہا ان کے میں قربان
بھیّا کہیں زخمی تو نہیں میرے چچا جان
اکبر نے کہا خیر سے ہیں عموئے ذیشان
یہ سنتے ہی دوڑی گئی خیمہ میں وہ نادان
سب سے کہا مشکیزہ بھرے لاتے ہیں گھر میں
لو صاحبو دریا سے چچا آتے ہیں گھر میں

81
کیا انس ہے میں اپنے چچا جان کے واری
وہ روتے تھے جب کرتی تھی میں گریہ زاری
پانی نہ کسی اور کو دیتے کبھی ناری
تھا کام انھیں کا جو بھری مشک ہماری
پھیرا ہے خدا نے انھیں تلوار تلے سے
جلدی کہیں آویں تو لپٹ جاؤں گلے سے

82
کہتی تھی چچی سے کہ کہاں ہو ادھر آؤ
حق تم کو سہاگن رکھے آنسو نہ بہاؤ
آتے ہیں چچا زانوں سے گردن تو اٹھاؤ
چھاتی سے چچیرے میرے بھائی کو لگاؤ
پیاسوں کو تمہیں پانی پلانا ہو مبارک
دریا سے چچا جان کا آنا ہو مبارک

83
بچوں نے بھی جس وقت سنا آتے ہیں عباس
ہاتھوں میں لئے جام سکینہ کے گئے پاس
دکھلا کے زبانوں کو یہ کہتے تھے بصد یاس
بولا نہیں جاتا یہ اسدم ہے ہمیں پیاس
پیاسی ہو بہت پہلے تمہیں پیجیو پانی
جھوٹا بہن اپنا ہمیں تم دیجیو پانی

84
خوش خوش گئے پھر خیمہ کی ڈیوڑھی پر ونادان
پردے سے لگے جھانکنے بچے سوئے میدان
وہ پوچھتے تھے کیوں نظر آتے ہیں چچا جان
باقر نے کہا شور ہے کچھ نہر پہ اس آن
عباس تو معلوم نہیں ہوتے ہیں رن میں
دادا وہ کمر پکڑے کھڑے روتے ہیں رن میں

85
فق ہو گئی اس بات کو سنتے ہی وہ دلگیر
گھبرا کے کہا خیر کرے مالک تقدیر
ناگاہ ہوا شور پکارے کئی بے پیر
صد شکر کہ بے دست ہوا بازو سے شبّیر
نوفل نے بڑے صاحب شمشیر کو مارا
عباس کو کیا قتل کیا شیر کو مارا

86
 پہنچی جو سکینہ کو صدائے سپہ شام
سراپنا لگی پیٹنے ہاتھوں سے وہ ناکام
ناموس کے خیمہ میں بھی برپا ہوا کہرام
گھرائے یہ معصوم کہ ہاتھوں سے گرے جام
بے آس تھے منھ چاند سا ایک ایک کافق تھا
سینوں میں جگر کا نپتے تھے دلکو قلق تھا

87
منھ دیکھ کے مظلوم سکینہ کا بصد یاس
چلاتے تھے فریاد ہے مارے گئے عباس
یاپانی کی امید تھی یا ٹوٹ گئی آس
اب کون ترس کھا کے بجھاویگا بھلا پیاس
مشکیزہ چھدے بازو بر اک تیغ سے کٹ جائے
وہ کیا کریں پیاسوں کی جو تقدیر الٹ جائے

88
بچوں کا یہ عالم تھا یہ تھی حالت سرور
چلاتے تھے ہے ہے میرے غمخوار بر اور
روتے تھے کمر باپ کی تھامے ہوئے اکبر
شہ کہتے تھے لاشے پہ مجھے لے چلو دلبر
آواز بھی رن سے میرا بھائی نہیں دیتا
اکبر مجھے آنکھونسے دکھائی نہیں دیتا

89
یہ کہ کے سوئے نہر چلے سید ابرار
ایک ہاتھ پسر تھامے تھا تھی ایک میں تلوار
جا پہونچے ترائی میں جو با دیدۂ خونبار
دیکھا کہ لب ۔۔۔۔۔۔ ہے بے ہوش علمدار
بازو ہیں جدا تن سے زرہ خونمیں بھری تھی
نیزے سے چھدی مشک کلیجہ پہ دھری تھی

90
عباس نے پائی گل زہرہ کی جو خوشبو
حسرت سے نظر کی طرفِ سید خوش خو
بولا نہ گیا آنکھوں سے بہنے لگے آنسو
منھ ملنے لگا پاؤں پہ حضرت کے وہ مہرو
جو حق غلامی تھا ادا کر گئے عباس
رکھ کر قدم شاہ پہ سر مر گئے عباس

91
سرپیٹ کے ہاتھوں سے پکارے شہ عالم
ہے ہے مرے جعفر مرے حمزہ مرے رستم
ہے ہے میرے عاشق میرے یا ور میرے ہمدم
ہے ہے میرے صفدر میرے حیدر میرے ضیغم
کون اب لڑے تلوار لئے ساتھ ہمارے
تم کیا ہوئے بے دست کٹے ہاتھ ہمارے

92
خامہ کو بس اب روک انیس جگر افگار
ہر بیت ہے اس مرثیے کی مطلعٔ انوار
برگشتہ زمانہ ہے تو کچھ غم نہیں زنہار
حامی ہیں ترے حضرت عباس علمدار
فیاض در عالم ہیں انیس غربا ہیں
سب مشکلیں آسان ہیں کہ وہ عقدہ کشا ہیں
