مرثیہ میر انیس

1
مسجد میں قتل جب شہ خیبر شکن ہوئے
زینت دو سریر امامت حسن ہوئے
سبط رسولؐ جب کہ امام زمن ہوئے
ممنون فیض عام سے سب مرد وزن ہوئے
چرچا تھا خوبیوں کا فلک سے زمیں تلک
خلق حسنؑ کا شور تھا عرش بریں تلک

2
حلات مشکلات امیر و فقیر تھے
اور شاہ لافتیٰ کے معظم وزیر تھے
قائم مقام سرور گردوں سریر تھے
کیا گوہر نسب میں عدیم النظیر تھے
نانا رسولؐ بادسشہ مشرقین سا
بازو اگر دیا تو خدا نے حسینؑ سا

3
لکھتا ہے یوں بھار میں راویٔ خوش کلام
اک دن سوار جاتے تھے شاہ ذو الاحترام
خدام تھے سواری میں سر گرم اہتمام
جھکتے تھے صف بہ صف پئے تسلیم خاص و عام
تھی روشنی زیاد تجلی طور سے
رستے بھرے تھے نور الٰہی کے نور سے

4
ناگاہ آیا سامنے اک مرد خیره سر
اور جانب امام درشتی سے کی نظر
منھ سے کلام سخت کہے اس نے بیشتر
ترک ادب ہے لاؤں اسے کس زبان پر
سمجھا نہ رتبہ شہ عالی مقام کو
دشنام دی امام علیہ السلام کو

5
جب کر چکا وہ ہے ادبانہ یہ سب کلام
حضرت نے اس کا دیکھ کے منھ روک لی لگام
اور مسکرا کے آپ نے کی سبقت کلام
فرمایا کیوں ہے غیظ میں اے مرد نیک نام
شاید اسیر دام بلاؤ محن ہے تو
مجھ کو گماں یہ ہے کہ غریب الوطن ہے تو

6
مجھ سے سوال کر کہ میں حاجت روا کروں
اور درد مفلسی ہو تو اس کی دوا کروں
گر تو مریض ہے تو شفا کی دعا کروں
مقروض گر تو ہووے تو اس کو ادا کروں
تنا ہے گر تو آ کے مرا غم گسار ہو
پیدل ہے گر تو گھوڑے پہ میرے سوار ہو

7
گر عازم سفر ہے تو لے مجھ سے زاد راہ
بھاگا ہے گر کہیں سے تو یاں آ کے لے پناہ
بھوکا ہے گر تو سیر ہو اے بندۂ الٰہ
عریاں ہے گر تو لے یہ مرا جامہ و کلاہ
اہل کرم ہیں لال جناب امیر کے
مہمان چند روز ہو گھر میں فقیر کے

8
حاضر رہوں گا میں تری خدمت میں روز و شب
ہر گز کسی طرح کا نہ ہوگا مجھے تعب
مہماں کو دوست رکھتے ہیں یاں خاصگان رب
دے گا حسینؑ تجھ کو زر و مال بے طلب
ہم آل مصطفےٰ ہیں یہ سب گھر رحیم ہے
مجھ سے زیادہ میرا برادر کریم ہے

9
سن کر کلام باد شہ آسماں سریر
کانپا مشال بید سراپا وہ مرد پیر
بے اختیار رو کے پکارا کہ اے قدیر
تیرا کوئی عدیل نہ اس کا کوئی نظیر
شیر خدا وصیٔ نبیؐ لا کلام ہے
حقا کہ تو امام ہے ابن امام ہے

10
حیدؑر سے بغض تھا مجھے اور آپ سے عناد
مانند روح و جسم ہوا آج اتحاد
دل سے تمام محو ہوئے باطنی فساد
اب بخشیے خطا کو یہی ہے مری مُراد
تعزیر دیجے تیغ دو پیکر نکالیے
تقصیر دار ہوں میں زباں کاٹ ڈالیے

11
چھاتی لگا کے کہنے لگا وہ خدا کا نور
بیتاب کس لیے ہے ترا کچھ نہیں قصور
ایمان لایا تو مرے دل کو ہوا سرور
نزدیک تو بہشت سے ہے اور سقر سے دور
آل نبیؐ کی تجھ سے محبت زیاد ہو
تجھ سے حسنؑ خوشی ہے خدا تجھ سے شاد ہو

12
کیا پرورش تھی بندے پر اس حکم کے نثار
ایسا گناہگار ہوا دم میں رستگار
مولا کا علم و فیض ہے عالم پہ آشکار
ہے رحم بے حساب تو بخشش ہے بے شمار
پیدا ہوئے تھے یہ عوض خاک نور سے
رحمت سے دل بھرا تھا تن پاک نور سے

13
ہمت کو نام پاک کی نسبت سے تھا شرف
لعل و گہر سے بھردئے تھے سائلوں کے کف
پھرتے تھے کو چہ ہائے مدینہ میں جس طرف
ہو جاتے تھے فقیر غنی دم میں صف بہ صف
انبار نان دوش پہ اپنے اٹھاتے تھے
بھوکوں کو جا کے راتوں کو کھانا کھلاتے تھے

14
اخبار صادقین سے ہوتا ہے یہ عیاں
اسما تھی ایک زوجہ شہزادۂ زماں
اس کی طرف سے رہتے تھے مولا جو بدگماں
کہتے تھے لوگ اس کا بتا دیجئے نشاں
فرماتے تھے گریز نہیں ہے ممات سے
قطع حیات ہے شدنی اس کی بات سے

15
آخر ہوا وہ کہتے تھے جو شاہ نامدار 
شهد و رطب میں زہر دیا اس نے تین بار
درد شکم سے رہتے تھے نالاں و بیقرار
سو سو طرح کے رنج تھے اور ایک جان زار
کیا بیکسی تھی راحت جان بتول پر
جا جا کے لوٹتے تھے مزار رسولؐ پر

16
مصروف تھے خدا کی عبادت میں صبح و شام
گھر سے کہیں نہ جاتا تھا شہزادۂ انام
یثرب میں اک مکاں تھا بنا کردۂ امام
منظور ہو گیا شب رحلت و ہیں مقام
رشک بہشت غیرت باغ جناں ہوا
اس شب کو وہ مکان تو بس لامکاں ہوا

17
وہ سبط مصطفؐےٰ کی شہادت کی رات تھی 
آفت کی رات تھی وہ مصیبت کی رات تھی
عالم کے بادشاہ کی رحلت کی رات تھی
زہرؑا و مرتضیٰ پہ قیامت کی رات تھی
گذری قلق میں فاطمہؑ کے نور عین کو
حجر حسنؑ میں نیند نہ آئی حسینؑ کو

18
بستر پہ جلوہ گر جو ہوا وہ فلک جناب
تا نصف شب نہ چشم ہوئی آشنائے خواب
زیر زمیں تھا خاک بسر قرص آفتاب
داغ جگر دکھاتا تھا جھک جھک کے آفتاب
تارے زمیں پہ ٹوٹ کے پیہم گرا کئے
چشم فلک سے قطرہ شبنم گرا کئے

19
پڑھ کر نماز شب کو جو سوئے شہ اُمم
آسما نے پھر ملا دیا پانی میں آکے سم
زینبؑ کو چونک کر یہ پکارے بہ صد الم
بھینا ابھی گلے سے بنیؐ کے لگے تھے ہم
رقّت یہ تھی کہ اشکوں سے تر روئے پاک تھا
محبوب کبریا کا گریبان چاک تھا

20
نانا گلے لگا کے یہ کہتے تھے بار بار
اے بیکس و غریب حسنؑ میں ترے نثار
پھرتے تھے میرے گرد علیؑ کو نہ تھا قرار
اماں بلائیں لے لے کے کرتی تھیں مجھ کو پیار
پوچھا جو میں نے آپ کا کیوں زنگ زرد ہے
رو کر کہا کہ آج کلیجے میں درد ہے

21
فرما کے یہ حسنؑ نے اُٹھایا زمیں سے جام
پایا درست اس کو جو تھا مہر کا مقام
تھوڑا سا پانی پی کے پکار وہ تلخ کام
دوڑو بہن کہ کام ہمارا ہوا تمام
یہ کہتے کہتے زرد رخ پاک ہو گیا
جلاتے تھے کہ ہائے جگر چاک ہو گیا

22
دوڑیں جناب زینبؑ بے کسی برہنہ پا
دیکھا کہ لوٹتے ہیں بچھونے پہ مجتباؑ
بولی یہ سر کو پیٹ کے وہ غم کی مبتلا
ہے ہے بہن نثار ہو بھیا یہ کیا ہوا
کیا پھر کسی نے زہر وغا سے پلا دیا
کس نے مرے کلیجے پہ خنجر پھرا دیا

23
بولے امام آہ نہیں طاقتِ سخن
جس کی کوئی دوا نہیں وہ درد ہے بہن
لپٹی گلے سے بھائی کے رو کر وہ خستہ تن
یہ بیقرار تھے کہ سنبھلتے نہ تھے حسنؑ
اور آیا تھا جو زہر جگر میں امام کے
جھکتے تھے بار بار کلیجے کو تھام کے

24
تکیے لگا دئیے تھے حرم نے ادھر ادھر
بازو کو کوئی تھامتی تھی اور کوئی سر
فرمایا طشت لاؤ ہوا زہر کار گر
رہ رہ کے کاٹتا ہے کوئی تیغ سے جگر
سینے سے منھ میں ٹکڑے کلیجے کے آتے ہیں
نانا بلا گئے تھے سو دنیا سے جاتے ہیں

25
زینبؑ نے جلد لا کے رکھا سامنے لگن
ہاتھوں سے دل پکڑ کے جھکے سرور زمن
آئی جو قے تو کان جواہر بنا دہن
الماس کھا کے لعل اُگلنے لگے حسنؑ
رنگ زمّردی کا سبب سب عیاں ہوا
معراج کی حدیث کا مطلب بیاں ہوا

26
فرماتے تھے حسنؑ کہ بلاؤ حسینؑ کو
تھا دم بہ دم یہ حکم کہ لاؤ حسینؑ کو
بھائی کا حالِ زار سناؤ حسینؑ کو
ٹکڑے مرے جگر کے دکھاؤ حسینؑ کو
کہدو کہ جلد آئیے رحلت کا وقت ہے
سُن جائیے کچھ آ کے وصیت کا وقت ہے

27
فضّہ نے جا کے دی شہ ذی جاہ کو خبر
وار فنا سے آپ کے بھائی کا ہے سفر
دوڑے حسینؑ چاک گریباں برہنہ سر
دیکھا تڑپ رہے ہیں شہنشاہ بحر و بر
گرنے لگے زمیں پہ جگر غم سے پھٹ گیا
پھیلا کے ہاتھ بھائی سے بھائی لپٹ گیا

28
دکھلا کے طشت بنت علیؑ نے کیا مقال
قربان جاؤں دیکھو یہ ہے مجتبؑیٰ کا حال
امید اب نہیں کہ بچے فاطمہؑ کا لال
فریاد ہے تباہ ہوئی مصؐطفےٰ کی آل
اک آن درد سے نہیں بھائی سنبھلتے ہیں
رہ رہ کے اب کلیجے کے ٹکڑے نکلتے ہیں

29
بھائی کا حال دیکھ کے اُٹھا جگر میں درد
لوٹے زمیں پہ گر کے بھری گیسوؤں میں گرد
تر تھا بدن پسینے میں اور ہاتھ پاؤں سرد
ان کا تو رنگ سبز تھا اور ان کا رنگ زرد
رعشہ تھا تن میں آنکھوں سے آنسو نکلتے تھے
جھک جھک کے منھ کو بھائی کے قدموں سے ملتے تھے

30
ہر دم لپٹ کے بھائی سے با چشم اشکبار
چلاتے تھے کہ آپ کی غربت کے میں نثار
شپیرؑ گلے کو چوم کے کہتے تھے بار بار
میں تجھ پہ صدقے اے مرے نانا کی یادگار
مجھ سے زیادہ ظلم و ستم تم پہ ہوئیں گے
ہم قبرمیں تمہاری مصیبت کو روئیں گے

31
بھیّا تمہاری گود میں نکلے گا میرا دم
تم دو گے ہم کو غسل و کفن جب مریں گے ہم
ہوگا تمہارے پاس نہ کوئی بجز الم
قاتل سر ہائے ہوئے گا یا خنجر ستم
شپّر تو بعد مرگ کے راحت سے سوئے گا
لاشہ تمہارا گھوڑوں سے پامال ہوئے گا

32
مرتا ہوں یاں وطن میں پہ تم ہوگے بے وطن
مرنے کے بعد بھی نہ ملے گا تمھیں کفن
پالوگے تم یتیموں کو میرے بصد محن
بچے تمہارے ہوئیں گے وابستہ رسن
جاویں گے اہل بیت نبیؐ شہر شام میں
سر ننگے ہوں گی بیبیاں بلوائے عام میں

33
کوزہ اُٹھا کے بولے حسینؑ فلک جناب
دیکھوں تو پی کے میں کہ یہ کس طرح کا ہے آب
بھائی کو لپٹی دوڑ کے زینبؑ جگر کباب
تھاما حسنؑ نے دست بر اور بصد شتاب
جلدی سے پی نہ لے یہ قلق تھا امام پر
اک ہاتھ تھا کلیجے پہ اک ہاتھ جام پر

34
کوزے کو لے کے بھائی سے پھینکا بروئے خاک
اتنی زمین ہو گئی گرتے ہی چاک چاک
رو کر کہا کرو گے اگر آپ کو ہلاک
کیوں کر پلیں گے میرے یتیمان درد ناک
تم آب تیغ پی کے زمانے سے جاؤ گے
دو دن کی پیاس خنجر کیں سے بجھاؤ گے

35
عباسؑ کو بلا کے کہا اے وفا شعار
چھٹپن سے تم کو کرتا ہے شپیؑر دل سے پیار
اور تم بھی ان کے نام پہ سو جاں سے ہو نثار
حاجت تو کچھ نہیں ہے سفارش کی زینہار
تم شاہ بیکساں کی رفاقت نہ چھوڑیو
بھائی کا اپنے دامن دولت نہ چھوڑیو

36
وقت جہاد معرکہ آرائی کیجیو
پیاری نہ جان بھائی سے اے بھائی کیجیو
لے کر علم کو فوج کی زیبائی کیجیو
پیاسے ہوں اہلِ بیت تو سقائی کیجیو
اب ہم تو سوئے خلد بریں جانے والے ہیں
تم ان کے ساتھ ہو یہ تمھارے حوالے ہیں

37
بندوں کی طرح قدموں پہ آقا کا سر رہے
بھائی پہ تیر آئیں تو سینہ سپر رہے
زینبؑ کی پردہ داری کی ہر دم خبر رہے
شفقت کی اس بھتیجے پہ ہردم نظر رہے
بیوہ کا لخت دل ہے نہ یہ دھیان کیجیو
قاسمؑ کو پہلے بھائی پہ قربان کیجیو

38
قاسمؑ کو پیار کر کے کیا اس طرح کلام
اس گھر کے تم چراغ ہو روشن ہے تم سے نام
ہیں بر سر سفر شہ مظلوم صبح و شام
کیجو وہ کام جس میں رضامند ہو امام
پیارے شہید تیغ جفا ہو کے آئیو
فردوس میں چچا پہ فدا ہو کے آئیو

39
مجمل یہ تھا جو تم نے سنا آج میری جاں
پہونچو گے کربلا میں تو ہو جائے گا عیاں
لکھ کر کچھ اپنے ہاتھ سے با چشم خوں فشاں
بازو پہ اس کے باندھ کے پھر یہ کیا بیاں
اس کو سوائے وقت مصیبت نہ کھولیو
یہ حرز جاں ہے غیر ضرورت نہ کھولیو

40
تھا ایک شاہزادوں میں فرزند خورد سال
حاضر حضور شاہ میں تھا وہ بھی نونہال
ہاتھوں کو اس کے چوم کے کرتے تھے یوں مقال
اے آخری شہید خوشا تجھ پسر کا حال
نازک کلائیاں یہ تری دل سے بھائی ہیں
حوریں ابھی سے گود میں لینے کو آئی ہیں

41
یہ کہتے کہتے غش ہوئے شاہ فلک جناب
غمگین تو تھے حسینؑ ہوا اور اضطراب
حاضر تھی روح احمدؐ و زہرؑا و بوترابؑ
وا تھے برنگ دیدۂ نرگس جناں کے باب
تشریف خلد کو شہ ذی جاہ لے گئے
جدّو پدر جو آئے تھے ہمراہ لے گئے

42
ماتم کی اہلِ بیت رسالت میں تھی صدا
برپا تھا شور واحَسَنؑا وامحمدؐا
سادات کے محلے میں اک حشر تھا بپا
بام فلک سے آتی تھی ہاتف کی یہ ندا
ال نبیؐ سے سبط محمدؐ جدا ہوئے
زہرؑا کے آج لعل و زمرّد جدا ہوئے

43
برپا تھا اہل بیت محمدؐ میں شور و شین
بہنیں پچھاڑیں خاک پہ کھاتی تھیں کر کے بین

اب اٹھ گیا زمانے سے ہم بیکسوں کا چین
بازو ہمارا ٹوٹ گیا وامصیبتا
باپ آج ہم سے چھوٹ گیا وامصیبتا

44
لاشے کے پاس مادر قاسمؑ کا تھا یہ حال
رخ زرد اور پھٹا تھا گریباں کھلے تھے بال

45

تھے شیثؑ و نوحؑ چاک گریباں لٹے ہوئے
الیاسؑ و خضرؑ جاتے تھے کاندھا دیے ہوئے

46
عباسؑ و عونؑ زیدؑ و محمدؐ تھے نوجہ گر

اور تھیں پس جنازہ آقائے خوش سیر
ازواج و خادمات محل سب برہنہ سر
عریاں سروں پہ مریؑم و حوّاؑ کے ہاتھ تھے
جبریلؑ پا برہنہ جنازے کے ساتھ تھے

47
جس دم جنازۂ پسر شاہ لافتاہ

یہ دشمنوں پہ مطلب شپیرؑ کھل گیا
یعنی قریب مرقد سلطان آنبیؑا
زینبؑ کنار شاہ رسل کے مزار ہو
منظور ہے کہ حق کو بہ مرکز قرار ہو

48
از بس کہ نسل فاطمہؑ سے تھا انہیں عناد
بس مستعد وہ ہو گئے سب بر سر فساد

یہ گہر ہماری ملک کا ہے اور مال کا
حصہ نہیں ہے اس میں محمدؐ کی آل کا

49
برہم ہوئے یہ سن کے امام فلک جناب
تھرایا جسم غیظ سے آنکھیں ہوئیں پُر آب
فرمایا دو گے حشر میں کیا حق کو تم جواب
ہے کچھ بھی پاس احمدؐ و زہرؑا و بوترابؑ
اوروں کی قبر ہا ئے بنیؐ کے قریب ہو
پہلو رسولؐ کا نہ حسنؑ کو نصیب ہو

50
عباسؑ نامدار کو بھی آگیا جلال
اور بھائی بند ہو گئے آمادۂ جدال
قبضوں پہ ہاتھ رکھے تھے شیر خدا کے لال
کہتے تھے ہم سے لڑ سکیں کیا ان کی ہے مجال
ان باغیوں کے زور کو دم بھر میں توڑیں گے
ہمسایۂ رسولِؐ خدا ہم نہ چھوڑیں گے

51
اک اک کو روکتے تھے امام فلک سریر
آنے لگے جنازے پہ جو اس طرف سے تیر
کانپی زمین ہل گیا صدمے سے چرخ پیر
نکلی لحد سے فاطمہؑ جیب کفن کو چیر
تابوت پر نواسے کے بدعت جو ہوتی تھی
روح رسولِؐ پاک جنازے کو روتی تھی

52
تابوت پر جو آنے لگے تیر ناگہاں
آماده نبرد ہوئے شاہ انس و جاں
آئی ندائے غیب کہ اے شاہ بیکساں
سر پیٹتی ہے فاطمہ زہرؑا بصد فغاں
جنبش میں اس الم سے محمدؐ کی قبر ہے
اے شیر حق کے لال یہ ہنگامِ صبر ہے

53
واں دی کسی نے زینبؑ بے کس کو یہ خبر
چلتے ہیں تیر ظلم حسن کے جنازے پر
تیغیں کھنچیں ہیں غصے میں ہیں شاہ بحر و بر
اٹھ کر لگی وہ پیٹنے ہاتھوں سے اپنا سر
چلائی گھر سے مضطرب الحال جاتی ہوں
قبر نبیؐ پہ کھولے ہوئے بال جاتی ہوں

54
زہرؑا کے گھر میں شور قیامت بپا ہوا
ڈیوڑھی پہ آئی دختر حیدؑر برہنہ پا
بکھرے ہوئے تھے بال گریبان تھا پھٹا
ایک ایک سے یہ کہتی تھی آنسو بہا بہا
غصہ ہے آج فاطمہؑ کے نور عین کو
للہ کوئی جاکے سنبھالو حسینؑ کو

55
زخمی کہیں نہ ہو پسر ضیغم الہٰ
ہم بیکسوں کا اب وہی وارث وہی ہے شاہ
اک بھائی کو تو قتل کیا دشمنوں نے آہ
کچھ ان پہ بن گئی تو یہ گھر ہو گیا تباہ
ابتو حسنؑ سے تا بہ قیامت جدائی ہے
ہے ہے یہی جہاں میں مرا ایک بھائی ہے

56
جا کر کہے جو کوئی کہ اے شاہ نامدار
صدقہ نبیؐ کی روح کا کیجو نہ کار زار
بابا کا صبر یاد کرو تم پہ میں نثار
ہے ہے نہ مجتبیٰ کا بناؤ وہاں مزار
آرام دو حبیب خدا کے حبیب کو
لیجا کے ماں کے پہلو میں گاڑو غریب کو

57
رو کر بیاں یہ کرتی تھی یاں زینبؑ حزیں
واں لے گئے بقیعہ میں تابوت شاہ دیں
مہر سپہر دیں ہوا پنہاں تہ زمیں
رویا لپٹ کے قبر سے زہرؑا کا نازنیں
محشر بپا تھا نالہ و فریاد و آہ سے
ہفتاد تیر نکلے تھے تابوت شاہ سے

58
بھائی کو دفن کر کے وہ شاہ فلک سریر
روضے پہ مصطفیٰ کے ہوئے جا کے گوشہ گیر
اُن کے بھی قتل کرنے کو درپے ہوئے شریر
آماده سفر ہوا کونین کا امیر
راحت کا اس شریف کی نقشہ بگڑ گیا
آباد کربلا ہوئی یشرب اُجڑ گیا 

59
پہونچے جو دشت ماریہ میں چھوڑ کر وطن
تھوڑے دنوں میں جا کے اُٹھائے بہت محن
بے آب و دانہ تھے حرم سرور زمن
مضطر شب دہم کو رہی شاہ کی بہن
بچپن سے انس تھا جو امام امم کے ساتھ
جاتی تھی جان آمد فوج ستم کے ساتھ

60
حسرت سے دیکھتی تھی کبھی شاہ کا جمال
بانو کے رانڈ ہونے کا تھا دم بدم خیال
چادر گری تھی فرق سے بکھرے تھے سر کے بال
دل پر گرا تھا شاہ کے کوہ غم و ملال
یاد خدا میں سبط رسولؐ ہمام کو
تھا انتظار صبح شہ تشنہ کام کو

61

زینبؑ کے لعل برچھیاں کھا کھا کے مر گئے

62
